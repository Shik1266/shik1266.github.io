  <script type="module">
    import { initializeApp } from "https://www.gstatic.com/firebasejs/12.15.0/firebase-app.js";
    import { getAuth, signInAnonymously, onAuthStateChanged } from "https://www.gstatic.com/firebasejs/12.15.0/firebase-auth.js";
    import { getFirestore, collection, addDoc, getDocs } from "https://www.gstatic.com/firebasejs/12.15.0/firebase-firestore.js";

    // 💡 [중요] 선생님의 개인 Firebase(git-io-33713) 설정값 적용 완료!
    let firebaseConfig = {
      apiKey: "AIzaSyBnjCWfcbvrvadmQ3bjoRJTD-rXSKzp_Bo",
      authDomain: "git-io-33713.firebaseapp.com",
      projectId: "git-io-33713",
      storageBucket: "git-io-33713.firebasestorage.app",
      messagingSenderId: "488733080531",
      appId: "1:488733080531:web:2a95f7c25856b078ce8d26",
      measurementId: "G-58XJP9R6HB"
    };

    // 현재 캔버스(테스트) 환경 지원용 코드 (유지)
    const firebaseConfigStr = typeof __firebase_config !== 'undefined' ? __firebase_config : '{}';
    try { 
        const parsed = JSON.parse(firebaseConfigStr);
        if(Object.keys(parsed).length > 0) firebaseConfig = parsed;
    } catch(e) {}
    
    const appId = typeof __app_id !== 'undefined' ? __app_id : 'demo-family-app';

    if (Object.keys(firebaseConfig).length > 0) {
        const app = initializeApp(firebaseConfig);
        const auth = getAuth(app);
        const db = getFirestore(app);
        let currentUser = null;

        // 익명으로 자동 로그인 처리 (접속 기록을 남기기 위함)
        const initAuth = async () => {
            await signInAnonymously(auth);
        };
        initAuth();

        onAuthStateChanged(auth, (user) => {
            currentUser = user;
            // 최초 접속 시 방문 기록 남기기 (1회만)
            if (user && !window.hasLoggedInitialVisit) {
                window.hasLoggedInitialVisit = true;
                window.logVisitToDb("👀 누군가 페이지 접속함");
            }
        });

        // 글로벌 스코프에서 호출할 수 있도록 window 객체에 할당 (DB 쓰기 함수)
        window.logVisitToDb = async function(roleName) {
            if (!currentUser) return;
            
            // 📱 핸드폰 기종(OS) 파악 로직 추가
            let deviceInfo = "PC/기타";
            const ua = navigator.userAgent;
            if (/iPhone|iPad|iPod/i.test(ua)) deviceInfo = "아이폰/아이패드";
            else if (/Android/i.test(ua)) deviceInfo = "안드로이드(갤럭시 등)";
            else if (/Mac/i.test(ua)) deviceInfo = "Mac PC";
            else if (/Windows/i.test(ua)) deviceInfo = "Windows PC";

            try {
                const logsCol = collection(db, 'artifacts', appId, 'public', 'data', 'visitLogs');
                await addDoc(logsCol, {
                    role: roleName,
                    device: deviceInfo, // 접속 기기 정보 저장
                    timestamp: new Date().toISOString()
                });
            } catch (e) {
                console.error("Log error", e);
            }
        };

        // 글로벌 스코프에서 호출할 수 있도록 window 객체에 할당 (DB 읽기 함수 - 관리자용)
        window.loadVisitLogsFromDb = async function() {
            const listEl = document.getElementById('visit-log-list');
            if (!currentUser) {
                listEl.innerHTML = '<li class="text-xs text-red-500 p-3 text-center">데이터베이스 접속 대기 중이거나 실패했습니다.</li>';
                return;
            }
            listEl.innerHTML = '<li class="text-xs text-slate-500 p-4 text-center"><i class="fa-solid fa-spinner fa-spin mr-2"></i>비밀 기록을 불러오는 중...</li>';

            try {
                const logsCol = collection(db, 'artifacts', appId, 'public', 'data', 'visitLogs');
                const snapshot = await getDocs(logsCol);
                let logs = [];
                snapshot.forEach(doc => logs.push(doc.data()));

                // 최신 순으로 정렬
                logs.sort((a, b) => new Date(b.timestamp) - new Date(a.timestamp));

                listEl.innerHTML = '';
                if (logs.length === 0) {
                    listEl.innerHTML = '<li class="text-xs text-slate-500 p-4 text-center">아직 접속 기록이 없습니다.</li>';
                    return;
                }

                logs.forEach(log => {
                    const date = new Date(log.timestamp);
                    const isGPS = log.role.includes("위치공유");
                    // 보기 편한 시간 형식으로 변환
                    const timeStr = date.toLocaleString('ko-KR', { month: 'short', day: 'numeric', hour: '2-digit', minute:'2-digit', second:'2-digit' });
                    const deviceStr = log.device ? ` / 📱 ${log.device}` : ''; // 기기 정보 표시

                    const li = document.createElement('li');
                    li.className = 'border-b border-slate-100 py-2 px-3 text-xs flex flex-col justify-center bg-white rounded shadow-sm mb-1.5 transition hover:bg-slate-50';

                    let icon = isGPS ? '<i class="fa-solid fa-location-dot text-emerald-500 w-5 text-center"></i>' : '<i class="fa-regular fa-eye text-slate-400 w-5 text-center"></i>';

                    li.innerHTML = `
                        <div class="flex items-center gap-2 mb-0.5">
                            ${icon}
                            <span class="font-bold ${isGPS ? 'text-emerald-700' : 'text-slate-600'}">${log.role}</span>
                        </div>
                        <div class="text-[10px] text-slate-400 font-mono tracking-tighter pl-7">
                            ${timeStr}${deviceStr}
                        </div>
                    `;
                    listEl.appendChild(li);
                });
            } catch (e) {
                listEl.innerHTML = `<li class="text-xs text-red-500 p-3">데이터 로딩 오류: ${e.message}</li>`;
            }
        }
    } else {
        // 실제 Github 환경 등 DB 세팅이 안 된 곳을 위한 예외 처리
        window.logVisitToDb = function() { console.log("DB not configured."); };
        window.loadVisitLogsFromDb = function() {
            document.getElementById('visit-log-list').innerHTML = '<li class="text-xs text-red-500 p-4 leading-relaxed">현재 캔버스 임시 DB가 연결되지 않았습니다.<br>개인 Github 환경에서 접속 기록을 보시려면 본인의 Firebase 설정값을 위 코드(firebaseConfig)에 붙여넣어 주세요!</li>';
        };
    }

      function updateSlidePosition() {

      }

      function updateDotsState() {
        let activeDotIndex = currentSlideIndex - 1;
        if (currentSlideIndex === 0) activeDotIndex = originalLength - 1;
        if (currentSlideIndex === originalLength + 1) activeDotIndex = 0;

        for (let i = 0; i < dots.length; i++) {

        }
      }

      function goToSlide(index, animate = true) {
        if (isSlideTransitioning) return;
        isSlideTransitioning = true;

        if (animate) {
          track.style.transition = 'transform 300ms cubic-bezier(0.25, 0.46, 0.45, 0.94)';
        } else {
          track.style.transition = 'none';
        }

        currentSlideIndex = index;
        updateSlidePosition();
        updateDotsState();
      }

      track.addEventListener('transitionend', () => {
        isSlideTransitioning = false;
        if (currentSlideIndex === 0) {
          track.style.transition = 'none';
          currentSlideIndex = originalLength;
          updateSlidePosition();
        } else if (currentSlideIndex === originalLength + 1) {
          track.style.transition = 'none';
          currentSlideIndex = 1;
          updateSlidePosition();
        }
      });

      prevBtn.onclick = () => {
        if (isSlideTransitioning) return;
        resetAutoPlay();
        goToSlide(currentSlideIndex - 1);
      };

      nextBtn.onclick = () => {
        if (isSlideTransitioning) return;
        resetAutoPlay();
        goToSlide(currentSlideIndex + 1);
      };

      // 스와이프 및 드래그 동작
      let startX = 0;
      let currentX = 0;
      let isDragging = false;
      let startTranslate = 0;

      function getEventX(e) {
        return e.touches ? e.touches[0].clientX : e.clientX;
      }

      function onDragStart(e) {
        if (isSlideTransitioning) return;
        isDragging = true;
        startX = getEventX(e);
        track.style.transition = 'none';
        const viewportWidth = document.getElementById('carousel-viewport').offsetWidth;
        startTranslate = -currentSlideIndex * viewportWidth;
        resetAutoPlay();
      }

      function onDragMove(e) {
        if (!isDragging) return;
        currentX = getEventX(e);
        const diffX = currentX - startX;
        track.style.transform = `translateX(${startTranslate + diffX}px)`;
      }

      function onDragEnd() {
        if (!isDragging) return;
        isDragging = false;
        const viewportWidth = document.getElementById('carousel-viewport').offsetWidth;
        const diffX = currentX - startX;
        const threshold = viewportWidth * 0.15;

        if (Math.abs(diffX) > threshold) {
          if (diffX > 0) {
            goToSlide(currentSlideIndex - 1);
          } else {
            goToSlide(currentSlideIndex + 1);
          }
        } else {
          goToSlide(currentSlideIndex);
        }
      }

      track.addEventListener('touchstart', onDragStart, { passive: true });
      track.addEventListener('touchmove', onDragMove, { passive: true });
      track.addEventListener('touchend', onDragEnd);

      track.addEventListener('mousedown', onDragStart);
      window.addEventListener('mousemove', onDragMove);
      window.addEventListener('mouseup', onDragEnd);

      function startAutoPlay() {
        autoSlideInterval = setInterval(() => {
          goToSlide(currentSlideIndex + 1);
        }, 4500);
      }

      function resetAutoPlay() {
        clearInterval(autoSlideInterval);
        startAutoPlay();
      }

      updateSlidePosition();
      updateDotsState();
      startAutoPlay();
    }

    async function fetchKmaGangneungWeather() {
      const lat = 37.7518;
      const lon = 128.8761;
      
      const apiUrl = `https://api.open-meteo.com/v1/forecast?latitude=${lat}&longitude=${lon}&daily=weathercode,temperature_2m_max,temperature_2m_min,precipitation_probability_max&timezone=Asia%2FSeoul`;

        day1: { text: "맑고 화창함 ☀️", icon: "fa-solid fa-sun text-amber-500 animate-pulse", temp: "21°C / 28°C", pop: "강수확률: 10%" },

      };

      try {
        const response = await fetch(apiUrl);
        if (!response.ok) throw new Error("KMA Link Node Error");
        const data = await response.json();
        
        const dates = data.daily.time;
        const targetDay1 = "2026-06-21";
        const targetDay2 = "2026-06-22";
        
        const idx1 = dates.indexOf(targetDay1);
        const idx2 = dates.indexOf(targetDay2);
        
        if (idx1 !== -1 && idx2 !== -1) {
          const code1 = data.daily.weathercode[idx1];
          const maxTemp1 = Math.round(data.daily.temperature_2m_max[idx1]);
          const minTemp1 = Math.round(data.daily.temperature_2m_min[idx1]);
          const pop1 = data.daily.precipitation_probability_max[idx1];
          
          const code2 = data.daily.weathercode[idx2];
          const maxTemp2 = Math.round(data.daily.temperature_2m_max[idx2]);
          const minTemp2 = Math.round(data.daily.temperature_2m_min[idx2]);
          const pop2 = data.daily.precipitation_probability_max[idx2];
          
          applyKmaData(1, code1, minTemp1, maxTemp1, pop1);
          applyKmaData(2, code2, minTemp2, maxTemp2, pop2);

          useKmaBaseline(kmaStaticBaseline);
        }
      } catch (error) {
        useKmaBaseline(kmaStaticBaseline);
      }

    function connectMqtt() {
      try {
        const clientId = "client_" + Math.random().toString(36).substring(2, 15);

          onFailure: onMqttConnectFailure,
          keepAliveInterval: 60,

        };

    function onMqttMessageArrived(message) {
      if (isDemoMode) return; 
      try {
        const topic = message.destinationName;
        const senderRole = topic.replace(FAMILY_CHANNEL_ROOT + "/", "");
        
        if (senderRole === currentRoleName) return;

        if (message.payloadString === "") {
          if (familyMarkers[senderRole]) {
            liveMap.removeLayer(familyMarkers[senderRole]);
            delete familyMarkers[senderRole];
          }
          delete activeMembers[senderRole];
          updateActiveMembersListUI();
          return;
        }

        const payload = JSON.parse(message.payloadString);
        const lat = parseFloat(payload.lat);
        const lng = parseFloat(payload.lng);
        const lastSeen = payload.time;

        if (isNaN(lat) || isNaN(lng)) return;

        activeMembers[senderRole] = { lat, lng, time: lastSeen };
        updateActiveMembersListUI();

        if (familyMarkers[senderRole]) {
          familyMarkers[senderRole].setLatLng([lat, lng]);
        } else {
          let initialText = senderRole.substring(0, 2);
          const familyIcon = L.divIcon({
            className: 'pulse-marker bg-[#6f4e37] text-white flex items-center justify-center font-bold text-[10px] shadow-lg border-2 border-[#eed6cb]',
            html: `<div class="w-8 h-8 flex items-center justify-center">${initialText}</div>`,
            iconSize: [32, 32],
            iconAnchor: [16, 16]
          });

          const fMarker = L.marker([lat, lng], { icon: familyIcon }).addTo(liveMap);
          fMarker.bindPopup(`<div class="text-xs font-semibold"><b>${senderRole}</b><br>실시간 전송 중! (${lastSeen})</div>`);
          familyMarkers[senderRole] = fMarker;
        }
      } catch (e) {
        console.error("데이터 동적 가공 중 오류 발생: ", e);
      }
    }

    function startDemoSimulation() {
      isDemoMode = true;
      document.getElementById("sandbox-warning-banner").classList.remove("hidden");
      document.getElementById("status-text").innerText = "⚠️ 데모 시뮬레이션 모드 가동 중";
      
      if (!myMarker) {
        updateMyOnMapLocation(37.75, 128.87);
      }
      
      demoTimer = setInterval(() => {
        const nowTime = new Date().toLocaleTimeString('ko-KR', { hour: '2-digit', minute: '2-digit' });
        
        for (const [name, pos] of Object.entries(demoPositions)) {
          pos.lat += (pos.targetLat - pos.lat) * pos.speed;
          pos.lng += (pos.targetLng - pos.lng) * pos.speed;
          
          activeMembers[name] = { lat: pos.lat, lng: pos.lng, time: nowTime };
          
          if (familyMarkers[name]) {
            familyMarkers[name].setLatLng([pos.lat, pos.lng]);
          } else {
            let initialText = name.substring(0, 2);
            const familyIcon = L.divIcon({
              className: 'pulse-marker bg-[#6f4e37] text-white flex items-center justify-center font-bold text-[10px] shadow-lg border-2 border-[#eed6cb]',
              html: `<div class="w-8 h-8 flex items-center justify-center">${initialText}</div>`,
              iconSize: [32, 32],
              iconAnchor: [16, 16]
            });
            
            const fMarker = L.marker([pos.lat, pos.lng], { icon: familyIcon }).addTo(liveMap);
            fMarker.bindPopup(`<div class="text-xs font-semibold"><b>${name}</b><br>시뮬레이션 구동 중! (${nowTime})</div>`);
            familyMarkers[name] = fMarker;
          }
        }
        updateActiveMembersListUI();
      }, 3000);
      
      startGpsTracking(); 
    }

    function updateActiveMembersListUI() {
      const listEl = document.getElementById("active-members-list");
      if (!listEl) return;
      listEl.innerHTML = "";
      
      const myItem = document.createElement("li");
      myItem.className = "flex items-center gap-1.5 text-emerald-600 font-bold";
      myItem.innerHTML = `<span class="h-1.5 w-1.5 rounded-full bg-emerald-500"></span> ${currentRoleName || '나'} (나)`;
      listEl.appendChild(myItem);

      for (const [name, info] of Object.entries(activeMembers)) {
        const li = document.createElement("li");
        li.className = "flex items-center gap-1.5 text-slate-600 truncate";
        li.innerHTML = `<span class="h-1.5 w-1.5 rounded-full bg-amber-500"></span> ${name} (${info.time})`;
        listEl.appendChild(li);
      }
    }

    function startGpsTracking() {
      if (navigator.geolocation) {
        watchId = navigator.geolocation.watchPosition(position => {
          const lat = position.coords.latitude;
          const lng = position.coords.longitude;

          updateMyOnMapLocation(lat, lng);

          if (!isDemoMode && mqttClient && mqttClient.isConnected()) {
            const messageObj = {
              lat: lat,
              lng: lng,
              time: new Date().toLocaleTimeString('ko-KR', { hour: '2-digit', minute: '2-digit' })
            };
            const message = new Paho.MQTT.Message(JSON.stringify(messageObj));
            message.destinationName = FAMILY_CHANNEL_ROOT + "/" + currentRoleName;
            message.qos = 0; 
            message.retained = true; 
            mqttClient.send(message);
          }
        }, err => {
          console.warn("GPS 신호 수신 대기 중입니다: ", err.message);
          if (!isDemoMode) {
            document.getElementById("status-text").innerText = "GPS 신호를 수집하는 중...";
          }
        }, {
          enableHighAccuracy: true, 
          timeout: 10000,
          maximumAge: 0
        });
      }
    }

      activeMembers = {};
